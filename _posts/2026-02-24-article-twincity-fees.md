faqs:
  - question: "臺北市民辦理冰存真的有免費嗎？"
    answer: "是的。依臺北市殯葬管理處最新公告，本市市民遺體冷藏「前 7 日」費用為 NT$0。自第 8 日起才開始按階梯計費（第 8 至 14 日每日 NT$400，第 15 日起每日 NT$800）。注意：7 日內完成火化者，另可申請火化費免收優惠。"
  - question: "新北市板殯的冷藏費如何計算？逾期真的會追溯計費嗎？"
    answer: "新北市市民若在進館後 7 日內完成火化，冷藏費全免；但若超過 7 日，系統會「從第 1 天起追溯計費」，也就是原本免費的那 7 天也要一起補算。例如：停放 10 天，不是只付 3 天的費用，而是全部 10 天每日 NT$400，共 NT$4,000。這是最常讓家屬誤解的計費陷阱，請務必注意。"
  - question: "如何判定符合「市民資格」？"
    answer: "只要逝者本人、配偶，或直系血親（父母、子女）任一人設籍於該市滿 6 個月以上，即可享有市民優惠價。設籍未滿 6 個月或無設籍者，均以「非市民」費率計費，通常為市民價 2 至 3 倍。"
  - question: "什麼是「北北桃火化互惠」？我符合嗎？"
    answer: "台北市、新北市、桃園市三市政府自 2026 年起完整實施互惠協議：設籍三市任一的亡者，可在三市任一公立火化場以市民優惠價 NT$2,000 火化，不須支付跨區加成費用。設籍其他縣市者不適用。"
  - question: "禮廳費用新北和台北哪個比較便宜？"
    answer: "整體而言新北較便宜。新北市丁級禮廳每三小時 NT$100，冷氣費另計每小時 NT$50；台北市懷愛館丙級禮廳每場（3 小時）NT$1,100，但包含冷氣。實際費用需依選用的禮廳等級（丁乙丙甲）與使用時數計算，祥安禮儀師可協助試算並選擇最適合的場館。"
  - question: "臺北市冰存第 7 日起計費，還是第 8 日？"
    answer: "第 8 日起計費。臺北市市民前 7 日（含進館當日計算）冷藏費 NT$0，自第 8 日起才開始計費（每日 NT$400），第 15 日起每日 NT$800。「第 7 天」本身仍屬免費範圍。"

        var resultBox  = document.getElementById('calcResult');
        var daysEl     = document.getElementById('calcDays');
        var amountEl   = document.getElementById('calcAmount');
        var breakEl    = document.getElementById('calcBreakdown');
        var noteEl     = document.getElementById('calcNote');
        var warnEl     = document.getElementById('calcWarn');

        /* 驗證 */
        if (!startVal || !endVal) {
            daysEl.innerHTML = '<span class="calc-error">請輸入進館日期與出殯日期。</span>';
            amountEl.innerHTML = breakEl.innerHTML = noteEl.innerHTML = warnEl.innerHTML = '';
            resultBox.style.display = 'block';
            return;
        }

        var start = new Date(startVal);
        var end   = new Date(endVal);
        var days  = Math.ceil((end - start) / 86400000);

        if (days <= 0) {
            daysEl.innerHTML = '<span class="calc-error">出殯日期必須晚於進館日期。</span>';
            amountEl.innerHTML = breakEl.innerHTML = noteEl.innerHTML = warnEl.innerHTML = '';
            resultBox.style.display = 'block';
            return;
        }

        var total    = 0;
        var breakdown = '';
        var warn      = '';

        /* ── 臺北市：直接扣除制 ── */
        if (loc === 'TP') {
            var freedays = 0;
            var mid = 0;
            var high = 0;

            for (var i = 1; i <= days; i++) {
                if (isCitizen) {
                    if (i <= 7)       { freedays++; }
                    else if (i <= 14) { mid++; total += 400; }
                    else              { high++; total += 800; }
                } else {
                    if (i <= 7)       { total += 400; }
                    else if (i <= 14) { total += 800; }
                    else              { total += 1600; }
                }
            }

            if (isCitizen) {
                var parts = [];
                if (freedays > 0) parts.push('第 1-' + freedays + ' 天：NT$0（免費）');
                if (mid > 0)      parts.push('第 8-' + (7 + mid) + ' 天：' + mid + ' 天 × NT$400 = NT$' + (mid * 400).toLocaleString());
                if (high > 0)     parts.push('第 15 天起：' + high + ' 天 × NT$800 = NT$' + (high * 800).toLocaleString());
                breakdown = parts.join('<br>');
            } else {
                var d1 = Math.min(days, 7);
                var d2 = Math.min(Math.max(days - 7, 0), 7);
                var d3 = Math.max(days - 14, 0);
                var p2 = [];
                if (d1 > 0) p2.push('第 1-' + d1 + ' 天：' + d1 + ' 天 × NT$400 = NT$' + (d1 * 400).toLocaleString());
                if (d2 > 0) p2.push('第 8-' + (7 + d2) + ' 天：' + d2 + ' 天 × NT$800 = NT$' + (d2 * 800).toLocaleString());
                if (d3 > 0) p2.push('第 15 天起：' + d3 + ' 天 × NT$1,600 = NT$' + (d3 * 1600).toLocaleString());
                breakdown = p2.join('<br>');
            }

        /* ── 新北市：限期免收 / 逾期追溯 ── */
        } else {
            if (isCitizen && days <= 7) {
                total = 0;
                breakdown = '7 日內完成火化，冷藏費全免（NT$0）。';
            } else {
                /* 逾期：從第 1 天起全部追溯計費 */
                for (var j = 1; j <= days; j++) {
                    var rate = isCitizen ? 400 : 800;
                    total += (j >= 15) ? (rate * 2) : rate;
                }
                var rateLabel = isCitizen ? 'NT$400' : 'NT$800';
                var rateHigh  = isCitizen ? 'NT$800' : 'NT$1,600';
                var d14 = Math.min(days, 14);
                var d15 = Math.max(days - 14, 0);
                var pp = [];
                pp.push('第 1-' + d14 + ' 天：' + d14 + ' 天 × ' + rateLabel + ' = NT$' + (d14 * (isCitizen ? 400 : 800)).toLocaleString());
                if (d15 > 0) pp.push('第 15-' + days + ' 天：' + d15 + ' 天 × ' + rateHigh + ' = NT$' + (d15 * (isCitizen ? 800 : 1600)).toLocaleString());
                breakdown = pp.join('<br>');

                if (isCitizen && days > 7) {
                    warn = '⚠ 追溯計費提醒：新北市逾 7 日停放，冷藏費從<strong>第 1 天起全部追溯</strong>，非只計算逾期天數。共 ' + days + ' 天費用均需支付。';
                }
            }
        }

        /* 輸出 */
        daysEl.innerHTML   = '<strong>停放天數：</strong>' + days + ' 天（進館至出殯日）';
        amountEl.innerHTML = '<strong>預估冷藏費：</strong><span class="calc-result__amount">NT$' + total.toLocaleString() + '</span>';
        breakEl.innerHTML  = '<strong>費用明細：</strong><br>' + breakdown;
        noteEl.innerHTML   = '※ 本試算為估算參考，實際費用以各場館當期公告為準。火化費、禮廳費等項目不包含於本試算內。';
        warnEl.innerHTML   = warn;

        resultBox.style.display = 'block';
        resultBox.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    };
}());
</script>
